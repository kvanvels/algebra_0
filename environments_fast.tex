% Stub replacement for environments.tex, used only by
% master_fastcheck.tex. Reproduces the same environment names, same
% shared counter, same {title}{label} calling convention, and same
% cleveref name registration as the real tcolorbox-based environments
% -- but with plain text output instead of colored/breakable boxes,
% which is what actually costs the time. Never used for the real book
% PDF, only for fast reference/syntax verification.

\newcounter{mycounter}[section]
\def\themycounter{\arabic{chapter}.\arabic{section}.\arabic{mycounter}}
\def\theHmycounter{\theHchapter.\theHsection.\arabic{mycounter}}

\newcounter{exercounter}[section]
\def\theexercounter{\arabic{chapter}.\arabic{section}.\arabic{exercounter}}
\def\theHexercounter{\theHchapter.\theHsection.\arabic{exercounter}}

\makeatletter
\renewcommand{\theequation}
{\arabic{chapter}.\arabic{section}.\arabic{equation}}
\@addtoreset{equation}{chapter}
\makeatother

\newenvironment{solution}
{\begin{proof}[Solution]}{\end{proof}}

\newenvironment{hint}{%
  \noindent\textbf{Hint:} \bgroup\itshape%
}{%
  \egroup\par%
}

% cleveref determines a label's *type* (the word it prints, e.g.
% "Definition" vs "Theorem") from whichever counter \refstepcounter
% was last called on -- not from \crefname's key alone. Several of
% these environments share one counter (#5, e.g. mycounter) purely for
% numbering, which would make cleveref print the same word for all of
% them if we \refstepcounter{#5} directly. So each environment gets
% its own private counter (#1, the environment name) that
% \refstepcounter fires on for cleveref's sake, redefined to *display*
% #5's value; #5 itself is still the one that actually advances.
\newcommand{\declarefastthm}[5]{%
  % #1 = env name, #2 = display name, #3 = crefname, #4 = Crefname,
  % #5 = counter to step
  \newcounter{#1}%
  \expandafter\renewcommand\csname the#1\endcsname{\csname the#5\endcsname}%
  \newenvironment{#1}[2]{%
    \stepcounter{#5}%
    \refstepcounter{#1}%
    \par\noindent\textbf{#2~\csname the#1\endcsname.}~##1%
    \def\@thislabel{##2}%
    \expandafter\ifx\@thislabel\@empty\else\label{##2}\fi
  }{\par}%
  \crefname{#1}{\textsc{#3}}{\textsc{#3} s}%
  \Crefname{#1}{\textsc{#4}}{\textsc{#4} s}%
}

\declarefastthm{thm}{Theorem}{theorem}{Theorem}{mycounter}
\declarefastthm{axm}{Axiom}{axiom}{Axiom}{mycounter}
\declarefastthm{lem}{Lemma}{lemma}{Lemma}{mycounter}
\declarefastthm{prop}{Proposition}{proposition}{Proposition}{mycounter}
\declarefastthm{coro}{Corollary}{corollary}{Corollary}{mycounter}
\declarefastthm{rem}{Remark}{remark}{Remark}{mycounter}
\declarefastthm{defn}{Definition}{definition}{Definition}{mycounter}
\declarefastthm{fact}{Fact}{fact}{Fact}{mycounter}
\declarefastthm{exer}{Exercise}{exercise}{Exercise}{exercounter}
\declarefastthm{exam}{Example}{example}{Example}{mycounter}
