% Stub replacement for fonts.tex, selected by \fastbuildtrue in
% master.tex. Skips unicode-math/fontspec entirely (lualatex/xelatex-
% only) so this variant can compile under plain pdflatex, which starts
% much faster (no OpenType font loading) and is only used for
% reference/syntax checking -- never for producing the actual book
% PDF.

% \mathscr and \coloneq/\eqcolon are unicode-math built-ins (no glyphs
% of their own -- unicode-math wires them up from the loaded math
% font); plain-LaTeX equivalents substituted here so fast-mode chapter
% text compiles unchanged.
\usepackage{mathrsfs}
\newcommand{\coloneq}{\mathrel{:=}}
\newcommand{\eqcolon}{\mathrel{=:}}

% unicode-math makes symbol macros like \lnot mode-tolerant (they
% auto-invoke math mode), which the book relies on by using \lnot bare
% as an exercise's referring-marker title (e.g.
% \begin{exer}{\lnot}{...}), outside any $...$. Plain amssymb's \lnot
% is strictly math-only and errors there, so make it math-tolerant too.
\renewcommand{\lnot}{\ensuremath{\neg}}
